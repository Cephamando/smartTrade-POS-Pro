### docker-compose.yml
```php
services:
  # 1. Apache + PHP (Bundled)
  web:
    container_name: pos_app
    build: 
      context: .
      dockerfile: docker/php/Dockerfile
    ports:
      - "${APP_PORT:-8090}:80"
    volumes:
      - ./:/var/www/html  # Maps the whole project root
    environment:
      APP_ENV: ${APP_ENV}
      DB_HOST: ${DB_HOST}
      DB_NAME: ${DB_NAME}
      DB_USER: root       # Using root for dev simplicity
      DB_PASS: ${DB_ROOT_PASS}
    depends_on:
      db:
        condition: service_started
    networks:
      - pos-net

  # 2. Database
  db:
    container_name: pos_db
    image: mysql:8.3
    restart: unless-stopped
    environment:
      MYSQL_DATABASE: ${DB_NAME}
      MYSQL_ROOT_PASSWORD: ${DB_ROOT_PASS}
      MYSQL_USER: ${DB_USER}
      MYSQL_PASSWORD: ${DB_PASS}
      MYSQL_ROOT_HOST: "%" # Allows remote root login
      TZ: ${TZ}
    ports:
      - "${DB_PORT_EXTERNAL:-8091}:3306"
    volumes:
      - db_data:/var/lib/mysql
      - ./docker/mysql/init.sql:/docker-entrypoint-initdb.d/init.sql
    networks:
      - pos-net

  # 3. phpMyAdmin
  phpmyadmin:
    image: phpmyadmin:latest
    container_name: pos_pma
    ports:
      - "${PHPMYADMIN_PORT:-8092}:80"
    environment:
      PMA_HOST: db
      PMA_PORT: 3306
      UPLOAD_LIMIT: 64M
    depends_on:
      - db
    networks:
      - pos-net

  # 4. Adminer
  adminer:
    image: adminer
    container_name: pos_adminer
    ports:
      - "${ADMINER_PORT:-8093}:8080"
    depends_on:
      - db
    networks:
      - pos-net

volumes:
  db_data:

networks:
  pos-net:
    driver: bridge
```

### docker/php/Dockerfile
```php
# Use 8.2 or 8.3 (8.5 is not yet standard in Docker Hub)
FROM php:8.3-apache

# Prevent interactive prompts during build
ARG DEBIAN_FRONTEND=noninteractive

# 1. Install System Dependencies
RUN apt-get update && apt-get install -y \
    libzip-dev zip unzip curl \
    libpng-dev libjpeg-dev libfreetype6-dev \
    libonig-dev libxml2-dev \
    --no-install-recommends \
    && rm -rf /var/lib/apt/lists/*

# 2. Configure & Install PHP Extensions
RUN docker-php-ext-configure gd --with-freetype --with-jpeg
RUN docker-php-ext-install -j$(nproc) \
    mysqli pdo pdo_mysql zip gd mbstring exif pcntl bcmath

# 3. Enable Apache Rewrite Module (Essential for modern PHP apps)
RUN a2enmod rewrite headers

# 4. CHANGE DOCUMENT ROOT to 'public'
# This points Apache to /var/www/html/public instead of /var/www/html
ENV APACHE_DOCUMENT_ROOT /var/www/html/public

RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/sites-available/*.conf
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/apache2.conf /etc/apache2/conf-available/*.conf

# 5. Security Configuration (Hide Server Info)
RUN echo "ServerTokens Prod\nServerSignature Off" >> /etc/apache2/apache2.conf

# 6. PHP Custom Config
RUN echo "memory_limit = 256M\n\
upload_max_filesize = 64M\n\
post_max_size = 64M\n\
display_errors = Off\n\
log_errors = On" > /usr/local/etc/php/conf.d/app.ini

# 7. Install Composer
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer

# 8. Set Working Directory
WORKDIR /var/www/html

# 9. Permissions (Ensure www-data owns the folder)
RUN chown -R www-data:www-data /var/www/html

```

### src/config.php
```php
<?php
// Prevent direct access
if (basename($_SERVER['PHP_SELF']) == basename(__FILE__)) { die('Access denied'); }

// DB Settings - Docker Aware
$host = getenv('DB_HOST') ?: 'pos_db';
$db   = getenv('DB_NAME') ?: 'pos_db';
$user = getenv('DB_USER') ?: 'root';
$pass = getenv('DB_PASS') ?: 'posRoot123!';
$charset = 'utf8mb4';

$dsn = "mysql:host=$host;dbname=$db;charset=$charset";
$options = [
    PDO::ATTR_ERRMODE            => PDO::ERRMODE_EXCEPTION,
    PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
    PDO::ATTR_EMULATE_PREPARES   => false,
];

try {
    $pdo = new PDO($dsn, $user, $pass, $options);
} catch (\PDOException $e) {
    error_log($e->getMessage());
    die("Database Connection Error. Please check logs.");
}

// Start Session Globally
if (session_status() === PHP_SESSION_NONE) {
    session_start();
}
?>
```
