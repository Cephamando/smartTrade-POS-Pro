<script>
    const urlParams = new URLSearchParams(window.location.search);
    if (urlParams.get('tab') === 'vendors') {
        new bootstrap.Tab(document.querySelector('#vendor-tab')).show();
    }

    // --- PRODUCT LOGIC ---
    const pModal = new bootstrap.Modal(document.getElementById('prodModal'));
    
    function openProductModal() {
        document.getElementById('p_id').value = ''; 
        document.getElementById('p_name').value = '';
        document.getElementById('p_sku').value = '';
        document.getElementById('p_price').value = '';
        document.getElementById('p_cost').value = '';
        document.getElementById('p_open').checked = false;
        document.getElementById('prodModalLabel').innerText = "Add New Product";
        pModal.show();
    }

    function editProduct(p) {
        document.getElementById('p_id').value = p.id; 
        document.getElementById('p_name').value = p.name;
        document.getElementById('p_sku').value = p.sku;
        document.getElementById('p_price').value = p.price;
        document.getElementById('p_cost').value = p.cost_price;
        document.getElementById('p_cat').value = p.category_id;
        document.getElementById('p_open').checked = p.is_open_price == 1;
        document.getElementById('prodModalLabel').innerText = "Edit Product";
        pModal.show();
    }

    function showHistory(pid) {
        Swal.fire({
            icon: 'info',
            title: 'Stock History',
            text: 'Detailed stock movement history feature coming soon!'
        });
    }

    // --- VENDOR LOGIC ---
    function editVendor(v) {
        document.getElementById('v_id').value = v.id;
        document.getElementById('v_name').value = v.name;
        document.getElementById('v_contact').value = v.contact_person;
        document.getElementById('v_phone').value = v.phone;
        document.getElementById('v_email').value = v.email;
        document.getElementById('vendorFormTitle').innerText = "Edit Vendor";
        document.getElementById('v_btn').innerText = "Update Vendor";
        document.getElementById('v_btn').classList.replace('btn-primary', 'btn-warning');
        document.getElementById('v_cancel').classList.remove('d-none');
    }
    
    function resetVendorForm() {
        document.getElementById('v_id').value = '';
        document.getElementById('v_name').value = '';
        document.getElementById('v_contact').value = '';
        document.getElementById('v_phone').value = '';
        document.getElementById('v_email').value = '';
        document.getElementById('vendorFormTitle').innerText = "Add New Vendor";
        document.getElementById('v_btn').innerText = "Save Vendor";
        document.getElementById('v_btn').classList.replace('btn-warning', 'btn-primary');
        document.getElementById('v_cancel').classList.add('d-none');
    }

    <?php if(isset($_SESSION['swal_msg'])): ?>
    Swal.fire({
        icon: <?= json_encode($_SESSION['swal_type']) ?>,
        title: <?= json_encode($_SESSION['swal_msg']) ?>,
        showConfirmButton: false, timer: 1500
    });
    <?php unset($_SESSION['swal_type'], $_SESSION['swal_msg']); ?>
    <?php endif; ?>
</script>
